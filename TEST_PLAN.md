## 测试脚本

### 创建实例测试脚本
```bash
#!/bin/bash

# OneClickVirt API测试脚本
API_URL="https://oneclickvirt.ypvps.com/api/v1"
TOKEN="YOUR_ADMIN_TOKEN"

# 测试创建实例
echo "测试1: 创建单个实例"
curl -X POST "$API_URL/admin/instances" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "test-container-001",
    "provider": "1",
    "image": "ubuntu:20.04",
    "cpu": 1,
    "memory": 512,
    "disk": 10,
    "bandwidth": 100
  }'

echo -e "\n\n测试2: 查看实例列表"
curl -X GET "$API_URL/admin/instances" \
  -H "Authorization: Bearer $TOKEN"
```

### 并发测试脚本
```bash
#!/bin/bash

# 并发创建测试
API_URL="https://oneclickvirt.ypvps.com/api/v1"
TOKEN="YOUR_ADMIN_TOKEN"

echo "开始并发测试..."
for i in {1..10}; do
  (
    curl -X POST "$API_URL/admin/instances" \
      -H "Authorization: Bearer $TOKEN" \
      -H "Content-Type: application/json" \
      -d "{
        \"name\": \"test-concurrent-$i\",
        \"provider\": \"1\",
        \"image\": \"ubuntu:20.04\",
        \"cpu\": 1,
        \"memory\": 512,
        \"disk\": 10,
        \"bandwidth\": 100
      }" &
  ) &
done

wait
echo "并发测试完成"
```
